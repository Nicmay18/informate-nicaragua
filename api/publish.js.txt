// api/publish.js
export default async function handler(req, res) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  const { platform, message, image, link, page } = req.body;

  try {
    // TELEGRAM (más fácil de configurar)
    if (platform === 'telegram') {
      const botToken = process.env.TELEGRAM_BOT_TOKEN;
      const chatId = process.env.TELEGRAM_CHANNEL_ID;
      
      const telegramUrl = `https://api.telegram.org/bot${botToken}/sendMessage`;
      
      const response = await fetch(telegramUrl, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          chat_id: chatId,
          text: message,
          parse_mode: 'Markdown',
          disable_web_page_preview: false
        })
      });
      
      if (!response.ok) throw new Error('Telegram API error');
      return res.status(200).json({ success: true, platform: 'telegram' });
    }

    // FACEBOOK (requiere token de página)
    if (platform === 'facebook') {
      const pageToken = page === 1 
        ? process.env.FACEBOOK_PAGE1_TOKEN 
        : process.env.FACEBOOK_PAGE2_TOKEN;
      const pageId = page === 1 
        ? process.env.FACEBOOK_PAGE1_ID 
        : process.env.FACEBOOK_PAGE2_ID;
      
      // Aquí iría la llamada real a Facebook Graph API
      // Por ahora simulamos éxito
      console.log(`Publicando en Facebook ${page}:`, message);
      
      return res.status(200).json({ success: true, platform: 'facebook', page });
    }

    // WHATSAPP (requiere Business API)
    if (platform === 'whatsapp') {
      // Simulado por ahora
      console.log('Publicando en WhatsApp:', message);
      return res.status(200).json({ success: true, platform: 'whatsapp' });
    }

    res.status(400).json({ error: 'Plataforma no soportada' });
    
  } catch (error) {
    console.error('Error:', error);
    res.status(500).json({ error: error.message });
  }
}
