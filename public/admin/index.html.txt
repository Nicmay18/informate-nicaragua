    function generarSlug(titulo) {
      return titulo.toLowerCase()
        .normalize('NFD').replace(/[\u0300-\u036f]/g, '')
        .replace(/[^a-z0-9]+/g, '-')
        .replace(/(^-|-$)/g, '') + '-' + Date.now().toString(36);
    }

    function formatearFecha(fecha) {
      if (!fecha) return "Fecha no disponible";
      if (fecha.toDate) {
        return fecha.toDate().toLocaleString("es-NI", { day: 'numeric', month: 'short', hour: '2-digit', minute: '2-digit' });
      }
      if (fecha.seconds) {
        return new Date(fecha.seconds * 1000).toLocaleString("es-NI", { day: 'numeric', month: 'short', hour: '2-digit', minute: '2-digit' });
      }
      return new Date(fecha).toLocaleString("es-NI", { day: 'numeric', month: 'short' });
    }

    async function cargarEstadisticas() {
      try {
        const noticiasSnap = await getDocs(collection(db, "noticias"));
        const totalNoticias = noticiasSnap.size;
        document.getElementById('statNoticias').textContent = totalNoticias;

        const conteo = {};
        categorias.forEach(c => conteo[c.id] = 0);
        noticiasSnap.forEach(d => {
          const cat = d.data().categoria;
          if (cat && conteo[cat] !== undefined) conteo[cat]++;
        });

        categorias.forEach(c => {
          const el = document.getElementById(`count${c.key}`);
          if (el) el.textContent = `${conteo[c.id] || 0} noticias`;
        });

        const statsRef = doc(db, "estadisticas", "visitas");
        const statsSnap = await getDoc(statsRef);
        document.getElementById('statVisitas').textContent = statsSnap.exists() ? 
          (statsSnap.data().total || 0).toLocaleString() : '0';

        document.getElementById('statHoy').textContent = Math.floor(Math.random() * 500 + 100);
        document.getElementById('statHoyVar').textContent = `+${Math.floor(Math.random() * 10)}`;
      } catch (error) {
        console.error("Error estadísticas:", error);
      }
    }

    async function cargarUltimasNoticias() {
      try {
        const q = query(collection(db, "noticias"), orderBy("fecha", "desc"));
        const snapshot = await getDocs(q);
        const noticias = [];
        snapshot.forEach(d => noticias.push({ id: d.id, ...d.data() }));

        const container = document.getElementById('ultimasNoticiasDashboard');
        if (noticias.length === 0) {
          container.innerHTML = '<p style="color: var(--gray); text-align: center;">No hay noticias</p>';
          return;
        }

        container.innerHTML = noticias.slice(0, 5).map(n => `
          <div class="news-item" style="padding: 12px 0;">
            <img src="${n.imagen || 'https://via.placeholder.com/80x60'}" class="news-thumb" alt="" onerror="this.src='https://via.placeholder.com/80x60?text=NI'">
            <div class="news-info">

        console.error("Error últimas noticias:", error);
      }
    }

    async function postToApi(endpoint, payload) {
      const resp = await fetch(`/api/${endpoint}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });
      const data = await resp.json();
      if (!resp.ok) throw new Error(data.error || data.message || `Error ${resp.status}`);
      return data;
    }

    window.guardarNoticia = async function(e) {
      e.preventDefault();
      showLoading(true);

      const statusDiv = document.getElementById('statusRedes');
      const statusText = document.getElementById('statusRedesTexto');
      statusDiv.style.display = 'block';
      statusText.textContent = 'Guardando noticia...';

      try {
        const noticiaId = document.getElementById('noticiaId').value;
        const titulo = document.getElementById('titulo').value.trim();
        const categoria = document.getElementById('categoria').value;
        const resumen = document.getElementById('resumen').value.trim();
        const contenido = document.getElementById('contenido').value.trim();
        const imagenUrl = document.getElementById('imagenUrl').value.trim();
        const enviarTelegram = document.getElementById('enviarTelegram').checked;
        const enviarFacebook = document.getElementById('enviarFacebook').checked;
        const destacada = document.getElementById('destacada').checked;

        if (!titulo || !categoria || !contenido || !imagenUrl) {
          showToast('Completa los campos obligatorios', 'error');
          showLoading(false);
          statusDiv.style.display = 'none';
          return;
        }

        const slug = noticiaId ? undefined : generarSlug(titulo);
        
        const noticiaData = {
          titulo,
          categoria,
          resumen: resumen || contenido.substring(0, 150) + '...',
          contenido,
          imagen: imagenUrl,
          fecha: serverTimestamp(),
          vistas: 0,
          publicadoPor: auth.currentUser?.email || 'admin',
          destacada,
          slug
        };

        let esNueva = !noticiaId;

        if (noticiaId) {
          noticiaData.actualizado = serverTimestamp();
          delete noticiaData.fecha;
          delete noticiaData.slug;
          await updateDoc(doc(db, "noticias", noticiaId), noticiaData);
          showToast('✅ Noticia actualizada', 'success');
        } else {
          const docRef = await addDoc(collection(db, "noticias"), noticiaData);
          noticiaData.id = docRef.id;
          showToast('✅ Noticia publicada', 'success');
        }

        // Enviar a redes SOLO si es nueva
        if (esNueva) {
          if (enviarTelegram) {
            statusText.textContent = 'Enviando a Telegram...';
            try {
              await postToApi('telegram', { noticia: noticiaData });
              showToast('📱 Enviado a Telegram', 'success');
            } catch (err) {
              showToast('Telegram: ' + err.message, 'error');
              console.error('Telegram error:', err);
            }
          }

          if (enviarFacebook) {
            statusText.textContent = 'Publicando en Facebook...';
            try {
              const fbResult = await postToApi('facebook', { noticia: noticiaData });
              showToast('📘 Publicado en Facebook', 'success');
              console.log('Facebook post:', fbResult);
            } catch (err) {
              showToast('Facebook: ' + err.message, 'error');
              console.error('Facebook error:', err);
            }
          }
        }

        limpiarFormulario();
        await cargarUltimasNoticias();
        await cargarEstadisticas();

      } catch (error) {
        console.error("Error guardando:", error);
        showToast('❌ Error: ' + error.message, 'error');
      } finally {
        showLoading(false);
        document.getElementById('statusRedes').style.display = 'none';
      }
    };

    window.cargarNoticias = async function(filtro = 'todas') {
      try {
        const q = query(collection(db, "noticias"), orderBy("fecha", "desc"));
        const snapshot = await getDocs(q);
        let noticias = [];
        snapshot.forEach(d => noticias.push({ id: d.id, ...d.data() }));

        if (filtro !== 'todas') {
          noticias = noticias.filter(n => n.categoria === filtro);
        }

        const container = document.getElementById('listaNoticiasCompleta');
        if (noticias.length === 0) {
          container.innerHTML = '<p style="color: var(--gray); text-align: center; padding: 40px;">No hay noticias publicadas</p>';
          return;
        }

        container.innerHTML = noticias.map(n => `
          <div class="news-item" data-id="${n.id}">
            <img src="${n.imagen || 'https://via.placeholder.com/80x60'}" class="news-thumb" alt="" onerror="this.src='https://via.placeholder.com/80x60?text=NI'">
            <div class="news-info">
              <div class="news-title">${n.titulo || '(Sin título)'}</div>
              <div class="news-meta">
                <span class="news-badge badge-${slugCategoria(n.categoria)}">${n.categoria || '—'}</span>
                <span><i class="fas fa-eye"></i> ${n.vistas || 0}</span>
                <span>${formatearFecha(n.fecha)}</span>
                ${n.destacada ? '<span style="color: gold;"><i class="fas fa-star"></i></span>' : ''}
                ${n.actualizado ? '<span style="color: var(--info);"><i class="fas fa-edit"></i></span>' : ''}
              </div>
            </div>
            <div class="news-actions">
              <button class="btn-icon edit" onclick="cargarNoticiaParaEditar('${n.id}')" title="Editar">
                <i class="fas fa-edit"></i>
              </button>
              <button class="btn-icon delete" onclick="eliminarNoticia('${n.id}')" title="Eliminar">
                <i class="fas fa-trash"></i>
              </button>
            </div>
          </div>
        `).join('');
      } catch (error) {
        console.error("Error cargando noticias:", error);
        showToast('Error cargando noticias: ' + error.message, 'error');
      }
    };

    window.cargarNoticiaParaEditar = async function(id) {
      showLoading(true);
      try {
        const docSnap = await getDoc(doc(db, "noticias", id));
        if (!docSnap.exists()) {
          showToast('No se encontró la noticia', 'error');
          return;
        }

        const data = docSnap.data();

        document.getElementById('noticiaId').value = id;
        document.getElementById('titulo').value = data.titulo || '';
        document.getElementById('categoria').value = data.categoria || '';
        document.getElementById('resumen').value = data.resumen || '';
        document.getElementById('contenido').value = data.contenido || '';
        document.getElementById('imagenUrl').value = data.imagen || '';
        document.getElementById('destacada').checked = !!data.destacada;

        actualizarPreview();

        document.getElementById('tituloFormulario').textContent = '✏️ Editar Noticia';
        document.getElementById('subtituloFormulario').textContent = 'Modificar Publicación';
        document.getElementById('btnGuardar').innerHTML = '<i class="fas fa-save"></i> Guardar Cambios';
        document.getElementById('btnCancelarEdicion').style.display = 'inline-flex';

        // No reenviar redes al editar
        document.getElementById('enviarTelegram').checked = false;
        document.getElementById('enviarFacebook').checked = false;

        showTab('nueva-tab', document.querySelectorAll('.nav-link')[1]);
        showToast('Noticia cargada para editar', 'success');
      } catch (error) {
        console.error("Error cargando noticia:", error);
        showToast('Error al cargar: ' + error.message, 'error');
      } finally {
        showLoading(false);
      }
    };

    window.eliminarNoticia = async function(id) {
      if (!confirm('¿Eliminar esta noticia permanentemente?')) return;

      showLoading(true);
      try {
        await deleteDoc(doc(db, "noticias", id));
        showToast('🗑️ Noticia eliminada', 'success');
        await cargarNoticias(window.categoriaActiva);
        await cargarUltimasNoticias();
        await cargarEstadisticas();
      } catch (error) {
        console.error("Error eliminando:", error);
        showToast('❌ Error al eliminar: ' + error.message, 'error');
      } finally {
        showLoading(false);
      }
    };

    window.buscarNoticias = function(termino) {
      const items = document.querySelectorAll('.news-item');
      items.forEach(item => {
        const titulo = item.querySelector('.news-title')?.textContent?.toLowerCase() || '';
        item.style.display = titulo.includes(termino.toLowerCase()) ? 'flex' : 'none';
      });
    };

    window.filtrarNoticias = function(categoria, btn) {
      document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
      if (btn) btn.classList.add('active');
      window.categoriaActiva = categoria;
      cargarNoticias(categoria);
    };

    window.filtrarPorCategoria = function(categoria) {
      showTab('noticias-tab', document.querySelectorAll('.nav-link')[2]);
      setTimeout(() => {
        document.querySelectorAll('.filter-btn').forEach(btn => {
          if (btn.textContent.includes(categoria)) btn.click();
        });
      }, 100);
    };

    async function cargarResumenCategorias() {
      try {
        const snapshot = await getDocs(collection(db, "noticias"));
        const conteo = {};
        categorias.forEach(c => conteo[c.id] = 0);
        snapshot.forEach(d => {
          const cat = d.data().categoria;
          if (conteo[cat] !== undefined) conteo[cat]++;
        });

        document.getElementById('resumenCategorias').innerHTML = categorias.map(c => `
          <div style="display:flex;justify-content:space-between;align-items:center;padding:12px 0;border-bottom:1px solid var(--light);">
            <div style="display:flex;align-items:center;gap:12px;">
              <div style="width:32px;height:32px;border-radius:8px;background:var(--light);display:flex;align-items:center;justify-content:center;">${c.emoji}</div>
              <span style="font-weight:600;">${c.id}</span>
            </div>
            <span style="font-weight:700;color:var(--primary);">${conteo[c.id]} noticias</span>
          </div>
        `).join('');

        document.getElementById('listaCategoriasConfig').innerHTML = categorias.map(c => `
          <div style="padding:12px;background:#f8fafc;border-radius:8px;display:flex;justify-content:space-between;align-items:center;margin-bottom:8px;">
            <span><i class="fas ${c.icon}" style="color: var(--${c.color}); margin-right: 8px;"></i> ${c.id}</span>
            <span class="news-badge badge-${c.color}">Activa</span>
          </div>
        `).join('');
      } catch (error) {
        console.error("Error categorías:", error);
      }
    }
